% Automatic tagging and document class
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\DocumentMetadata{
  tagging=on,
  pdfstandard=ua-2,
  lang=en-US,
  testphase={phase-III,math,table} % Required for improved structure support
}
\documentclass[12pt,a4paper]{article}
\tagpdfsetup{table/header-rows={1}} % Set default header to 1st row of tables
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% Metadata and hyperref
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\usepackage[
  pdfauthor={<Document Author>},
  pdftitle={<Document Title>}
]{hyperref}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% Layout and formatting packages
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\usepackage{tabularx,float,xcolor}
\usepackage[
  includehead,
  left=0.5in, right=0.5in,
  top=0.75in, bottom=0.75in,
  headheight=13.6pt
]{geometry}
